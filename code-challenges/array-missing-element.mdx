// Given an unsorted array of unique integers (size n + 1) and a first array identical to the second array but missing one integer (size n), find and output the missing integer.

// Examples:
// Input: list1 = [1, 2, 3, 4] list2 = [1, 2, 3, 4, 5] Output: 5

// Input: [], [1] Output: 1

// Constraints:
// -inf ≤ n ≤ inf
// All entries are unique
// length(list2) - length(list1) == 1

function findMissing(list1, list2) {
  assert(Array.isArray(list1), 'list1 must be an array')
  assert(Array.isArray(list2), 'list2 must be an array')
  assert(
    list2.length - list1.length === 1,
    'list2 must include one more element than list1'
  )
  const sum1 = list1.reduce((total, current) => total + current, 0)
  const sum2 = list2.reduce((total, current) => total + current, 0)

  return sum2 - sum1
}
